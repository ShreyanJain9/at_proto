def list(count = nil)
  (get_paginated_data(
    self.repo,
    "com.atproto.repo.listRecords",
    key: "records",
    params: { repo: self.repo.to_s, collection: collection },
  ) do |record|
    ATProto::Record.from_hash(record)
  end)
    .first(count) if count
end
